    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.2.3/dist/js/bootstrap.bundle.min.js"></script>
    <script>
        let data = {
            1: {name: "John", age: 30},
            2: {name: "Jane", age: 25}
        };
        let currentId = null;
        
        function renderData() {
            let tbody = document.getElementById("apidatatbody");
            tbody.innerHTML = "";
            for (let id in data) {
                let row = document.createElement("tr");
                row.innerHTML = `
                    <td>${id}</td>
                    <td>${data[id].name}</td>
                    <td>${data[id].age}</td>
                    <td>
                        <button class="btn btn-primary" onclick="updateHandler(${id})">Update</button>
                        <button class="btn btn-danger" onclick="deleteHandler(${id})">Delete</button>
                    </td>
                `;
                tbody.appendChild(row);
            }
        }
        
        function createHandler() {
            let name = document.getElementById("name").value;
            let age = document.getElementById("age").value;
            let newId = Object.keys(data).length + 1;
            data[newId] = {name: name, age: age};
            document.getElementById("name").value = "";
            document.getElementById("age").value = "";
            renderData();
        }
        
        function updateHandler(id) {
            currentId = id;
            document.getElementById("name").value = data[id].name;
            document.getElementById("age").value = data[id].age;
            document.getElementById("createbtn").style.display = "none";
            document.getElementById("updatebtn").style.display = "block";
            document.getElementById("deletebtn").style.display = "block";
        }
        
        function updateSaveHandler() {
            if (currentId !== null) {
                let name = document.getElementById("name").value;
                let age = document.getElementById("age").value;
                data[currentId].name = name;
                data[currentId].age = age;
                renderData();
                document.getElementById("createbtn").style.display = "block";
                document.getElementById("updatebtn").style.display = "none";
                document.getElementById("deletebtn").style.display = "none";
            }
        }
        
        function deleteHandler(id) {
            currentId = id;
            if (confirm(`Are you sure you want to delete ${data[id].name}?`)) {
                delete data[id];
                renderData();
            }
        }
        
        document.getElementById("createbtn").addEventListener("click", createHandler);
        document.getElementById("updatebtn").addEventListener("click", updateSaveHandler);
        document.getElementById("deletebtn").addEventListener("click", deleteHandler);
        
        renderData();
    </script>
